// DOM elements const raceContainer = document.getElementById('race-container'); const filterContainer = document.getElementById('filter-container'); const unitButtons = document.querySelectorAll('.unit-btn'); // Current state let currentUnit = 'km'; // Default unit is kilometers let currentFilter = 'all'; // Default filter is all races // Format date to be more readable function formatDate(dateString) { const options = { year: 'numeric', month: 'long', day: 'numeric' }; return new Date(dateString).toLocaleDateString(undefined, options); } // Determine if a race is a marathon function isMarathon(race) { return race.distanceKm === 42.2 || race.distanceMi === 26.2; } // Determine if a race is a half marathon function isHalfMarathon(race) { return race.distanceKm === 21.1 || race.distanceMi === 13.1; } // Format distance based on current unit function formatDistance(race) { // Special handling for marathon and half marathon if (isMarathon(race)) { return "Marathon"; } else if (isHalfMarathon(race)) { return "Half Marathon"; } else if (currentUnit === 'km') { return race.distanceKm === 5 || race.distanceKm === 10 ? `${race.distanceKm}K` : `${race.distanceKm} km`; } else { return `${race.distanceMi} mi`; } } // Helper to get all unique distances in current unit function getUniqueDistances() { const distances = raceData.map(race => formatDistance(race)); return [...new Set(distances)].sort((a, b) => { // Special sorting for Marathon and Half Marathon if (a === "Marathon") return 1; // Marathon should be last if (b === "Marathon") return -1; if (a === "Half Marathon") return 1; // Half Marathon before Marathon if (b === "Half Marathon") return -1; // For numeric distances, extract the number for sorting const numA = parseFloat(a); const numB = parseFloat(b); return numA - numB; }); } // Create filter buttons based on unique distances function createFilterButtons() { // Clear existing filter buttons except "All Races" filterContainer.innerHTML = ''; // Create the "All Races" button const allRacesButton = document.createElement('button'); allRacesButton.className = 'filter-btn'; allRacesButton.classList.add('active'); allRacesButton.setAttribute('data-distance', 'all'); allRacesButton.textContent = 'All Races'; allRacesButton.addEventListener('click', () => { setActiveFilter('all'); filterAndRenderRaces(); }); filterContainer.appendChild(allRacesButton); // Extract unique distances based on current unit const uniqueDistances = getUniqueDistances(); // Add filter buttons for each unique distance uniqueDistances.forEach(distance => { const button = document.createElement('button'); button.className = 'filter-btn'; button.setAttribute('data-distance', distance); button.textContent = distance; button.addEventListener('click', () => { setActiveFilter(distance); filterAndRenderRaces(); }); filterContainer.appendChild(button); }); } // Set active filter and update UI function setActiveFilter(distance) { currentFilter = distance; // Update active class on filter buttons document.querySelectorAll('.filter-btn').forEach(btn => { if (btn.getAttribute('data-distance') === distance) { btn.classList.add('active'); } else { btn.classList.remove('active'); } }); } // Filter races by distance and render them function filterAndRenderRaces() { let filteredRaces; if (currentFilter === 'all') { filteredRaces = [...raceData]; } else { // Filter races by the formatted distance label filteredRaces = raceData.filter(race => formatDistance(race) === currentFilter); } renderRaceCards(filteredRaces); } // Render race cards based on provided races array function renderRaceCards(races) { raceContainer.innerHTML = ''; if (races.length === 0) { raceContainer.innerHTML = `

No races found for the selected filter.

`; return; } // Sort races by date (newest first) const sortedRaces = [...races].sort((a, b) => new Date(b.date) - new Date(a.date) ); sortedRaces.forEach(race => { const card = document.createElement('div'); card.className = 'race-card'; // Add personal best class if it's a PB if (race.personalBest) { card.classList.add('pb'); } // Create race name HTML (with link if race website exists) const raceNameHTML = race.raceWebsite ? `

${race.name}

` : `

${race.name}

`; // Create HTML for race links (exclude race website since it's on the title now) let linksHTML = ''; if (race.resultsWebsite) { linksHTML += `Results`; } if (race.stravaActivity) { linksHTML += `Strava`; } // PB badge HTML const pbBadgeHTML = race.personalBest ? `
Personal Best
` : ''; // Format the distance with special labels const displayDistance = formatDistance(race); card.innerHTML = `
${formatDate(race.date)}
${raceNameHTML}
${displayDistance}
${pbBadgeHTML}
Time: ${race.time}
Pace: ${getPace(race)}
${linksHTML}
`; raceContainer.appendChild(card); }); } // Get pace based on current unit function getPace(race) { return currentUnit === 'km' ? race.paceKm : race.paceMi; } // Set up event listeners for unit toggle unitButtons.forEach(button => { button.addEventListener('click', () => { const unit = button.getAttribute('data-unit'); // Only process if unit is actually changing if (unit !== currentUnit) { // Update active class unitButtons.forEach(btn => btn.classList.remove('active')); button.classList.add('active'); // Update current unit currentUnit = unit; // Recreate filter buttons with new unit createFilterButtons(); // Reset to show all races when changing units setActiveFilter('all'); // Rerender races filterAndRenderRaces(); } }); }); // Initialize the page function init() { // Create filter buttons createFilterButtons(); // Initial render with all races filterAndRenderRaces(); } // Run initialization init();
